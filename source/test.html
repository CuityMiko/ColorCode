            var billTime = this.data.billInfo.time;
            if(this.data.stat == 'ok') {
                var timeDataHtml = '(' +billTime+ ')';
            } else {
                var timeDataHtml = billTime;
            }
            $A(this.node.query('.ui-box-title .module-statistics-time')).each(function(item, index){
